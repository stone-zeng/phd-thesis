\def\Tree#1#2#3#4#5#6{
  \def\x{0.6}
  \def\y{0.5}
  \begin{tikzpicture}[baseline=0.2cm, thick]
    \draw
      (0,0) -- (-3*\x,3*\y) node [above] {$a$}
      (#1)  -- (-\x,3*\y)   node [above] {$b$}
      (#2)  -- (\x,3*\y)    node [above] {$c$}
      (0,0) -- (3*\x,3*\y)  node [above] {$d$}
      (0,0) -- (0,-1.5*\y)  node [below] {$e$}
      (#3) node {\scriptsize $#5$}
      (#4) node {\scriptsize $#6$};
  \end{tikzpicture}
}
