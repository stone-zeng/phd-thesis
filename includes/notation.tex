\begin{notation}[lp{10em}]
  CFT   & 共形场论             \\ % conformal field theory
  DMRG  & 密度矩阵重整化群     \\ % density matrix renormalization group
  MERA  & 多尺度纠缠重整化方法 \\ % multiscale entanglement renormalization ansatz
  MPO   & 矩阵乘积算符         \\ % matrix product operator
  MPS   & 矩阵乘积态           \\ % matrix product state
  MTC   & 模张量范畴           \\ % modular tensor category
  PEPS  & 投影纠缠对态         \\ % projected entangled pair state
  SVD   & 奇异值分解           \\ % singular value decomposition
  TNR   & 张量网络重整化       \\ % tensor network renormalization
  TEBD  & 时间演化块消减       \\ % time-evolving block decimation
  TQFT  & 拓扑量子场论         \\ % topological quantum field theory
  TRG   & 张量重整化群         \\ % tensor renormalization group
  iMPS  & 无限 MPS             \\ % infinite MPS
  iTEBD & 无限 TEBD            \\ % infinite TEBD
  \\
  $\1$                                          & 单位对象      \\
  $d_i$                                         & 量子维数      \\
  $N_{ij}^k$                                    & 融合系数      \\
  $\bigl[F^{ijk}_l\bigr]_{xy}$, $F^{jik}_{lky}$ & $F$ 符号      \\
  $\mathbb{V}$                                  & 向量空间      \\
  $\chi$                                        & 连接维数      \\
  $\ket{\Psi}$                                  & 基态波函数    \\
  $\bra{\Omega}$                                & 直积态        \\
  $\ev{\Omega|\Psi}$                            & 奇异关联子    \\
  $\mathcal{T}_{pqr}^s$                         & 管状代数的基  \\
  $\mathcal{P}_i$                               & 中心幂等元    \\
  $L_n$, $\bar{L}_n$                            & Virasoro 算符 \\
  $J_n$, $\bar{J}_n$                            & Kac--Moody 算符
\end{notation}
